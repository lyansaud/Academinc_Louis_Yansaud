merge_data$Sex <- as.factor(merge_data$Sex)

merge_data %>%
  filter(!is.na(Age)) %>%
  group_by(Pclass) %>%
  summarise(Age_Pclass_mean = round(mean(Age)))

merge_data_missing_age <- merge_data %>%
                            filter(is.na(Age)) %>%
                            mutate(Age = ifelse(Pclass == "1",  39 ,

merge_data <- merge_data %>%
